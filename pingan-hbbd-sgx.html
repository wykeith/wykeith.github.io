Ping An¶

Founded in 1988, Ping An Insurance is an integrated financial service provider headquartered in Shenzhen. The company has a focus on the offerings of healthcare services and integrated financial products. Ping An is China's second-largest life and P&C; insurer. The company strives for an integrated financial services platform comprising life insurance, P&C; insurance, banking, other financial services, and technology. The group is known for its "finance + technology" strategy, heavily investing in technological innovation to transform traditional financial services.

import yfinance as yf
import matplotlib.pyplot as plt

# Define the ticker symbol
ticker_symbol = '2318.HK'  # Ping An Insurance ticker symbol on the Hong Kong Stock Exchange

# Workaround for yfinance issue with Yahoo Finance API
TICKERS = list()
TICKERS.append(ticker_symbol)
START_DATE = '2024-01-01'
END_DATE = '2025-03-23'
CACHED_DATA = dict()

data = yf.download(' '.join(TICKERS), start=START_DATE, end=END_DATE,
                   group_by='ticker')

tickers_to_retry = []

for ticker in TICKERS:
    download_success = [r for r in data[ticker]['Close'] if r > 0]
    if download_success:
        CACHED_DATA[ticker] = {'close': [r for r in data[ticker]['Close']],
                               'row_data': data[ticker],
                               'low': [r for r in data[ticker]['Low']],
                               'high': [r for r in data[ticker]['High']]}
    else:
        tickers_to_retry.append(ticker)

if tickers_to_retry:
    data = yf.download(' '.join(tickers_to_retry), start=START_DATE, end=END_DATE,
                       group_by='ticker')


for ticker in tickers_to_retry:
    download_success = [r for r in data[ticker]['Close'] if r > 0]
    if download_success:
        CACHED_DATA[ticker] = {'close': [r for r in data[ticker]['Close']],
                               'row_data': data[ticker],
                               'low': [r for r in data[ticker]['Low']],
                               'high': [r for r in data[ticker]['High']]}

# Plot the stock price
plt.figure(figsize=(10, 6))
plt.plot(CACHED_DATA[ticker_symbol]["row_data"].index, CACHED_DATA[ticker_symbol]['close'], label='Close Price')
plt.title(f'Ping An Group {ticker_symbol} Stock Price (Last 3 Month)')
plt.xlabel('Date')
plt.ylabel('Stock Price (HKD)')
plt.legend()
plt.grid(True)
plt.show()
